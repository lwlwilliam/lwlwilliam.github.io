下载依赖

从 modelscope 下载大模型镜像

from modelscope import snapshot_download
snapshot_download(model_id='google/gemma-4-E2B-it-qat-mobile-transformers', cache_dir='/Users/x/.cache/huggingface/hub')

运行大模型

from transformers import AutoModelForCausalLM, AutoTokenizer

model_id = '/Users/x/.cache/huggingface/hub/Qwen/Qwen3.5-0.8B'
model = AutoModelForCausalLM.from_pretrained(
    model_id,
    torch_dtype='auto',
    trust_remote_code=True,
)

tn = AutoTokenizer.from_pretrained(model_id)

prompt = ('用中文给起个艺名，姓『赵』。')

input_ids = tn.encode(prompt, return_tensors='pt')

generation_output = model.generate(
    input_ids=input_ids,
    max_new_tokens=10000
)
